#10.5
delta_0 = function(x)
{
  premierMembre=t(x)%*%solve(sigma)%*%mu0
  deuxiemeMembre=((t(mu0)%*%solve(sigma)%*%mu0)/2)[1]
  troisiemeMembre=log(pi0)
  return(premierMembre-deuxiemeMembre+troisiemeMembre)
}

delta_1 = function(x){
  premierMembre=t(x)%*%solve(sigma)%*%mu1
  deuxiemeMembre=((t(mu1)%*%solve(sigma)%*%mu1)/2)[1]
  troisiemeMembre=log(pi1)
  return(premierMembre-deuxiemeMembre+troisiemeMembre)
}
